# taken from brotli
if(NOT CMAKE_BUILD_TYPE)
    message(STATUS "Setting build type to Release as none was specified")
    set(CMAKE_BUILD_TYPE "Release" CACHE STRING "Choose the type of build" FORCE)
else()
    message(STATUS "Build type is '${CMAKE_BUILD_TYPE}'")
endif()

# For now, I'll be using the debug version of Jolt for both build versions
set(NOSTALGIA_JPH_DEFINITIONS JPH_PROFILE_ENABLED)

if(${CMAKE_BUILD_TYPE} STREQUAL Debug)
    set(NOSTALGIA_EXTERN_DIR_VER debug)
    set(NOSTALGIA_VERSION_DEFINITION NOSTALGIA_DEBUGGING)
else()
    set(NOSTALGIA_EXTERN_DIR_VER release)
    set(NOSTALGIA_VERSION_DEFINITION NOSTALGIA_RELEASE)
endif()

if(IS_WINDOWS)
    # For now, we're only using MinGW/MSYS to compile for Windows
    set(NOSTALGIA_EXTERN_DIR ${NOSTALGIA_ROOT}/static_libraries/windows-mingw)
else()
    set(NOSTALGIA_EXTERN_DIR ${NOSTALGIA_ROOT}/static_libraries/linux/${NOSTALGIA_EXTERN_DIR_VER})
endif()

# Because we're using MinGW/MSYS for the time being, static libs can be combined using
# the same method in both Linux and Windows
# https://stackoverflow.com/a/14200728
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libz.o
    COMMAND ${CMAKE_CXX_COMPILER} -r -o ${CMAKE_CURRENT_BINARY_DIR}/libz.o -Xlinker --whole-archive ${NOSTALGIA_EXTERN_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}z${CMAKE_STATIC_LIBRARY_SUFFIX})
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libbz2_static.o
    COMMAND ${CMAKE_CXX_COMPILER} -r -o ${CMAKE_CURRENT_BINARY_DIR}/libbz2_static.o -Xlinker --whole-archive ${NOSTALGIA_EXTERN_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}bz2_static${CMAKE_STATIC_LIBRARY_SUFFIX})
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libbrotlicommon.o
    COMMAND ${CMAKE_CXX_COMPILER} -r -o ${CMAKE_CURRENT_BINARY_DIR}/libbrotlicommon.o -Xlinker --whole-archive ${NOSTALGIA_EXTERN_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}brotlicommon${CMAKE_STATIC_LIBRARY_SUFFIX})
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libbrotlidec.o
    COMMAND ${CMAKE_CXX_COMPILER} -r -o ${CMAKE_CURRENT_BINARY_DIR}/libbrotlidec.o -Xlinker --whole-archive ${NOSTALGIA_EXTERN_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}brotlidec${CMAKE_STATIC_LIBRARY_SUFFIX})
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libbrotlienc.o
    COMMAND ${CMAKE_CXX_COMPILER} -r -o ${CMAKE_CURRENT_BINARY_DIR}/libbrotlienc.o -Xlinker --whole-archive ${NOSTALGIA_EXTERN_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}brotlienc${CMAKE_STATIC_LIBRARY_SUFFIX})
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libpng18.o
    COMMAND ${CMAKE_CXX_COMPILER} -r -o ${CMAKE_CURRENT_BINARY_DIR}/libpng18.o -Xlinker --whole-archive ${NOSTALGIA_EXTERN_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}png18${CMAKE_STATIC_LIBRARY_SUFFIX})
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libfreetype.o
    COMMAND ${CMAKE_CXX_COMPILER} -r -o ${CMAKE_CURRENT_BINARY_DIR}/libfreetype.o -Xlinker --whole-archive ${NOSTALGIA_EXTERN_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}freetype${CMAKE_STATIC_LIBRARY_SUFFIX})
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libglfw.o
    COMMAND ${CMAKE_CXX_COMPILER} -r -o ${CMAKE_CURRENT_BINARY_DIR}/libglfw.o -Xlinker --whole-archive ${NOSTALGIA_EXTERN_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}glfw${CMAKE_STATIC_LIBRARY_SUFFIX})
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libJolt.o
    COMMAND ${CMAKE_CXX_COMPILER} -r -o ${CMAKE_CURRENT_BINARY_DIR}/libJolt.o -Xlinker --whole-archive ${NOSTALGIA_EXTERN_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}Jolt${CMAKE_STATIC_LIBRARY_SUFFIX})
set(NOSTALGIA_OBJ_FILES
    ${CMAKE_CURRENT_BINARY_DIR}/libz.o
    ${CMAKE_CURRENT_BINARY_DIR}/libbz2_static.o
    ${CMAKE_CURRENT_BINARY_DIR}/libbrotlicommon.o

add_library(Nostalgia ${LIBRARY_VERSION} ${NOSTALGIA_SRC_FILES} ${NOSTALGIA_OBJ_FILES})
target_compile_features(Nostalgia PUBLIC cxx_std_26)
target_compile_definitions(Nostalgia PUBLIC
    ${NOSTALGIA_VERSION_DEFINITION}
    ${NOSTALGIA_JPH_DEFINITIONS})

if(${CMAKE_CXX_COMPILER_ID} EQUAL GNU OR ${CMAKE_CXX_COMPILER_ID} EQUAL MSVC)
    target_compile_definitions(Nostalgia PUBLIC NOSTALGIA_GCC_PCH)
endif()

target_precompile_headers(Nostalgia PUBLIC
    "$<$<COMPILE_LANGUAGE:CXX>:${NOSTALGIA_PCH_SRC}>"
    "$<$<COMPILE_LANGUAGE:C>:<stddef.h$<ANGLE-R>>")
target_include_directories(Nostalgia PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${NOSTALGIA_ROOT}
    ${NOSTALGIA_ROOT}/thirdparty
    ${NOSTALGIA_ROOT}/thirdparty/freetype/include)

set(EDITOR_ROOT ${CMAKE_CURRENT_SOURCE_DIR}/Editor)
set(EDITOR_SRC_FILES
    ${EDITOR_ROOT}/things/player.cpp
    ${EDITOR_ROOT}/app/nostalgia_goggles.cpp
    ${EDITOR_ROOT}/system/dedicated_main.cpp
    ${EDITOR_ROOT}/tools/stopwatch_log.cpp
    ${EDITOR_ROOT}/gui/imgui_debugger.cpp

if(BUILD_SHARED_LIBS)
    set(EDITOR_ENGINE_LIBRARY ${CMAKE_CURRENT_BINARY_DIR}/${CMAKE_SHARED_LIBRARY_PREFIX}Nostalgia${CMAKE_SHARED_LIBRARY_SUFFIX})
else()
    set(EDITOR_ENGINE_LIBRARY ${CMAKE_CURRENT_BINARY_DIR}/${CMAKE_STATIC_LIBRARY_PREFIX}Nostalgia${CMAKE_STATIC_LIBRARY_SUFFIX})
endif()

if(IS_WINDOWS)
    set(EDITOR_LIBRARIES_DIR ${EDITOR_ROOT}/libraries/windows)
else()
    set(EDITOR_LIBRARIES_DIR ${EDITOR_ROOT}/libraries/linux)
endif()

add_executable(NostalgiaGoggles ${EDITOR_SRC_FILES})
target_compile_features(NostalgiaGoggles PUBLIC cxx_std_26)
target_include_directories(NostalgiaGoggles
    PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}
    PUBLIC ${EDITOR_ROOT} ${EDITOR_ROOT}/thirdparty ${NOSTALGIA_ROOT}/thirdparty)
target_link_directories(NostalgiaGoggles PUBLIC ${CMAKE_CURRENT_BINARY_DIR} ${EDITOR_LIBRARIES_DIR})
target_link_libraries(NostalgiaGoggles PUBLIC getargs vulkan ${EDITOR_ENGINE_LIBRARY})

if(${CMAKE_BUILD_TYPE} STREQUAL Debug)
    if(NOT IS_WINDOWS AND ${CMAKE_CXX_COMPILER_ID} EQUAL Clang)
        target_compile_options(Nostalgia PUBLIC "-Wall -O0 -g -fsanitize=address")
        target_link_options(Nostalgia PUBLIC "-Wall -O0 -g -fsanitize=address")
        target_compile_options(NostalgiaGoggles PUBLIC "-Wall -O0 -g -fsanitize=address")
        target_link_options(NostalgiaGoggles PUBLIC "-Wall -O0 -g -fsanitize=address")
    endif()
endif()
